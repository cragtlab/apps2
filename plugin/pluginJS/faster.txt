(async function () {
  const container = PluginAPI.add("Faster");
  container.innerHTML = `
    <div >
      <button id="saveNotes">Save Notes</button>
      <button id="addTopic">Add Topic</button>
    </div>
    <section id="topic-list">
      <table border="1" style="width: 100%;">
        <thead>
          <tr>
            <th>Actions</th>
            <th>Topic</th>
            <th>Notes</th>
            <th>Quick Links</th>
          </tr>
        </thead>
        <tbody id="topics"></tbody>
      </table>
    </section>
  `;

  const rootHandle = PluginAPI.getRootHandle();
  const topics = [];

  async function readFile(name) {
    try {
      const file = await rootHandle.getFileHandle(name);
      const content = await (await file.getFile()).text();
      return JSON.parse(content);
    } catch (e) {
      console.error(`Failed to read ${name}:`, e);
      return null;
    }
  }

async function writeFile(name, contents) {
    const fileHandle = await rootHandle.getFileHandle(name, { create: true });
    const writable = await fileHandle.createWritable();
    await writable.write(JSON.stringify(contents));
    await writable.close();

    // Verify the file was saved correctly by reading it back
    const verifyHandle = await rootHandle.getFileHandle(name);
    const verifyFile = await verifyHandle.getFile();
    const verifyText = await verifyFile.text();

    let parsed;
    try {
        parsed = JSON.parse(verifyText);
    } catch (e) {
        throw new Error(`Save verification failed: file content is not valid JSON.`);
    }

    const written = JSON.stringify(contents);
    const readBack = JSON.stringify(parsed);
    if (written !== readBack) {
        throw new Error(`Save verification failed: written content does not match file on disk.`);
    }
}
  function renderTopics() {
    const topicList = container.querySelector('#topics');
    topicList.innerHTML = '';
    topics.forEach((topic, index) => {
      if (topic.snoozed) return;
      const row = document.createElement('tr');
      row.innerHTML = `
        <td>
          <button data-index="${index}" class="delete-topic">Delete</button>
          <button data-index="${index}" class="snooze-topic">Snooze</button>
        </td>
        <td contenteditable="true" data-index="${index}" data-field="name" class="editable">${topic.name}</td>
        <td contenteditable="true" data-index="${index}" data-field="notes" class="editable">${topic.notes}</td>
        <td>
          <ul>
            ${topic.links.map((link, linkIndex) => `
              <li>
                <a href="${link}" target="_blank">${link}</a>
                <button data-topic="${index}" data-link="${linkIndex}" class="delete-link">Delete</button>
              </li>`).join('')}
          </ul>
          <button data-index="${index}" class="add-link">Add Link</button>
        </td>
      `;
      topicList.appendChild(row);
    });
  }

  function promptAndAddTopic() {
    const name = prompt('Enter the topic name:').trim();
    if (name) {
      const notes = prompt('Enter the notes for this topic:').trim().replace(/\n/g, '<br>');
      topics.push({ name, notes, links: [], snoozed: false });
      renderTopics();
    }
  }

  function deleteTopic(index) {
    if (confirm(`Are you sure you want to delete the topic "${topics[index].name}"?`)) {
      topics.splice(index, 1);
      renderTopics();
    }
  }

  function snoozeTopic(index) {
    topics[index].snoozed = true;
    renderTopics();
  }

  function addLink(index) {
    const url = prompt('Enter the URL:');
    if (url) {
      topics[index].links.push(url);
      renderTopics();
    }
  }

  function deleteLink(topicIndex, linkIndex) {
    topics[topicIndex].links.splice(linkIndex, 1);
    renderTopics();
  }

  function updateTopic(index, newValue, field) {
    topics[index][field] = newValue.trim();
  }

 async function saveAllNotes() {
    let hasEmptyTopic = false;
    topics.forEach((topic, index) => {
        if (!topic.snoozed) {
            const row = container.querySelector(`#topics tr:nth-child(${index + 1})`);
            const nameCell = row.querySelector('td:nth-child(2)');
            const notesCell = row.querySelector('td:nth-child(3)');
            topic.name = nameCell.textContent.trim();
            topic.notes = notesCell.innerHTML.trim();
            if (!topic.name) hasEmptyTopic = true;
        }
    });

    if (hasEmptyTopic) {
        alert('Cannot save. One or more topics have empty names. Please fill in all topic names before saving.');
        return;
    }

    try {
        await writeFile('faster.json', { topics });

        const date = new Date();
        const hours = date.getHours().toString().padStart(2, '0');
        const minutes = date.getMinutes().toString().padStart(2, '0');
        const secs = date.getSeconds().toString().padStart(2, '0');
        document.title = `${hours}:${minutes}:${secs} Saved ✓`;

    } catch (e) {
        document.title = `Save FAILED`;
        //alert(`Save failed: ${e.message}`);
        throw e;
    }
}

  async function loadData() {
    try {
      const data = await readFile('faster.json');
      if (data && data.topics) {
        data.topics.forEach(topic => topics.push(topic));
        renderTopics();
        await writeFile('faster-backup.json', { topics });
      }
    } catch (error) {
      console.error('No data to load.'+error);
    }
  }

  container.querySelector('#saveNotes').onclick = async () => {
    try {
      await saveAllNotes();
    } catch (error) {
      //alert(error.message);
    }
  };

  container.querySelector('#addTopic').onclick = promptAndAddTopic;

  container.addEventListener('click', (e) => {
    if (e.target.classList.contains('delete-topic')) deleteTopic(Number(e.target.dataset.index));
    if (e.target.classList.contains('snooze-topic')) snoozeTopic(Number(e.target.dataset.index));
    if (e.target.classList.contains('add-link')) addLink(Number(e.target.dataset.index));
    if (e.target.classList.contains('delete-link')) deleteLink(Number(e.target.dataset.topic), Number(e.target.dataset.link));
  });

  container.addEventListener('blur', (e) => {
    if (e.target.classList.contains('editable')) {
      updateTopic(Number(e.target.dataset.index), e.target.innerHTML, e.target.dataset.field);
    }
  }, true);

async function autoSave() {
setInterval(async () => {
        try {
            await saveAllNotes();
        } catch (ignored) {
           // console.error('Auto-save failed:', error);
        }
    }, 300000);
}
  await loadData();
  autoSave();

  window.addEventListener('beforeunload', async (e) => {
    e.preventDefault();
    await saveAllNotes();
    e.returnValue = '';
  });

   
})();
